CAT01 - How to identify a cadastral parcel

Pregunta: ¿Cómo se identifica una parcela catastral?

Respuesta: Una parcela catastral se identifica mediante una referencia catastral única compuesta por 14 caracteres (7 dígitos de parcela + 7 dígitos de inmueble)

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?parcel ?parcelRef ?propertyRef WHERE {
    ?parcel a edintcat:CadastralParcel ;
            edintcat:hasCadastralReference ?ref .

    ?ref a edintcat:CadastralReference ;
         edintcat:parcelReference ?parcelRef ;
         edintcat:propertyReference ?propertyRef .
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?parcel ?propertyRef WHERE {
    ?parcel a edintcat:CadastralParcel ;
            edintcat:hasCadastralReference ?ref .

    ?ref a edintcat:CadastralReference ;
         edintcat:parcelReference "0569302" ;
         edintcat:propertyReference ?propertyRef .
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?parcel ?parcelRef ?propertyRef ?completeRef WHERE {
    ?parcel a edintcat:CadastralParcel ;
            edintcat:hasCadastralReference ?ref .

    ?ref a edintcat:CadastralReference ;
         edintcat:parcelReference ?parcelRef ;
         edintcat:propertyReference ?propertyRef .

    BIND(CONCAT(?parcelRef, ?propertyRef) AS ?completeRef)
}
CAT02 - What entities compose the Spanish Cadastre

Pregunta: ¿Qué entidades componen el catastro español?

Respuesta: El catastro español se compone de tres entidades principales: Parcelas Catastrales, Inmuebles Catastrales y Construcciones Catastrales

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?entityType (COUNT(?entity) AS ?count) WHERE {
    {
        ?entity a edintcat:CadastralParcel .
        BIND("Parcela Catastral" AS ?entityType)
    } UNION {
        ?entity a edintcat:CadastralProperty .
        BIND("Inmueble Catastral" AS ?entityType)
    } UNION {
        ?entity a edintcat:CadastralConstruction .
        BIND("Construcción Catastral" AS ?entityType)
    }
}
GROUP BY ?entityType

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?parcel ?parcelLabel (COUNT(?property) AS ?propertyCount) WHERE {
    ?parcel a edintcat:CadastralParcel ;
            rdfs:label ?parcelLabel ;
            edintcat:hasCadastralProperty ?property .
}
GROUP BY ?parcel ?parcelLabel

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?property ?propertyLabel (COUNT(?construction) AS ?constructionCount) WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?propertyLabel ;
              edintcat:hasCadastralConstruction ?construction .
}
GROUP BY ?property ?propertyLabel

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?parcel ?property ?construction ?propertyLabel ?constructionLabel WHERE {
    ?parcel a edintcat:CadastralParcel ;
            edintcat:hasCadastralProperty ?property .

    ?property a edintcat:CadastralProperty ;
             rdfs:label ?propertyLabel ;
             edintcat:hasCadastralConstruction ?construction .

    ?construction a edintcat:CadastralConstruction ;
                 rdfs:label ?constructionLabel .
}
CAT03 - What information does a cadastral property contain

Pregunta: ¿Qué información contiene un inmueble catastral?

Respuesta: Un inmueble catastral contiene: referencia catastral, porcentaje de participación, uso, clasificación, y opcionalmente dirección

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX callejero: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/callejero#>

SELECT ?property ?label ?participation ?parcelRef ?propertyRef WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?label ;
              edintcat:participation ?participation ;
              edintcat:hasCadastralReference ?ref .

    ?ref a edintcat:CadastralReference ;
         edintcat:parcelReference ?parcelRef ;
         edintcat:propertyReference ?propertyRef .
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX callejero: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/callejero#>

SELECT ?property ?label ?use ?class ?participation WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?label ;
              edintcat:hasUse ?use ;
              edintcat:hasClass ?class ;
              edintcat:participation ?participation .

    ?use a skos:Concept ;
         skos:prefLabel ?useLabel .

    ?class a skos:Concept ;
           skos:prefLabel ?classLabel .
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX callejero: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/callejero#>

SELECT ?property ?label ?streetName ?number ?postalCode ?municipality ?province WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?label ;
              edintcat:hasAddress ?address .

    ?address a callejero:Portal ;
             callejero:via ?street ;
             callejero:numero ?number ;
             callejero:codigoPostal ?postalCode ;
             callejero:municipio ?municipality ;
             callejero:provincia ?province .

    ?street a callejero:Via ;
             callejero:nombreVia ?streetName .
    }

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX callejero: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/callejero#>

SELECT ?property ?label ?participation ?useLabel ?classLabel
       ?parcelRef ?propertyRef ?streetName ?number ?postalCode WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?label ;
              edintcat:participation ?participation ;
              edintcat:hasUse ?use ;
              edintcat:hasClass ?class ;
              edintcat:hasCadastralReference ?ref ;
              OPTIONAL { ?property edintcat:hasAddress ?address } .

    ?use skos:prefLabel ?useLabel .
    ?class skos:prefLabel ?classLabel .

    ?ref edintcat:parcelReference ?parcelRef ;
         edintcat:propertyReference ?propertyRef .

    OPTIONAL {
        ?address callejero:numero ?number ;
                 callejero:codigoPostal ?postalCode ;
                 callejero:via ?street .
        ?street callejero:nombreVia ?streetName .
    }
}
CAT04 - How constructions relate to parcels

Pregunta: ¿Cómo se relacionan las construcciones con las parcelas?

Respuesta: Las construcciones se relacionan con las parcelas a través de un inmueble intermedio: Parcela → Inmueble → Construcción

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?construction ?constructionLabel ?property ?propertyLabel
       ?parcel ?parcelLabel WHERE {
    ?construction a edintcat:CadastralConstruction ;
                 rdfs:label ?constructionLabel ;
                 edintcat:onCadastralProperty ?property .

    ?property a edintcat:CadastralProperty ;
              rdfs:label ?propertyLabel ;
              edintcat:onCadastralParcel ?parcel .

    ?parcel a edintcat:CadastralParcel ;
             rdfs:label ?parcelLabel .
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?construction ?constructionLabel ?property ?propertyLabel
       ?parcel ?parcelRef WHERE {
    ?parcel a edintcat:CadastralParcel ;
            edintcat:hasCadastralReference ?parcelRefObj ;
            edintcat:hasCadastralProperty ?property .

    ?property edintcat:hasCadastralConstruction ?construction .

    ?parcelRefObj a edintcat:CadastralReference ;
                  edintcat:parcelReference "0569302" .

    ?construction rdfs:label ?constructionLabel .
    ?property rdfs:label ?propertyLabel .
    }

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?parcel ?parcelLabel (COUNT(?construction) AS ?constructionCount) WHERE {
    ?parcel a edintcat:CadastralParcel ;
            rdfs:label ?parcelLabel ;
            edintcat:hasCadastralProperty ?property ;
            edintcat:hasCadastralProperty/^edintcat:hasCadastralConstruction ?construction .
}
GROUP BY ?parcel ?parcelLabel

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?construction ?property ?parcel ?parcelRef ?propertyRef WHERE {
    ?construction a edintcat:CadastralConstruction ;
                 edintcat:onCadastralProperty ?property .

    ?property a edintcat:CadastralProperty ;
              edintcat:onCadastralParcel ?parcel ;
              edintcat:hasCadastralReference ?propRef .

    ?parcel edintcat:hasCadastralReference ?parcelRef .

    ?propRef edintcat:parcelReference ?parcelRefStr ;
             edintcat:propertyReference ?propertyRefStr .

    BIND(?parcelRefStr AS ?parcelRef)
    BIND(?propertyRefStr AS ?propertyRef)
}
CAT05 - What uses can a property have

Pregunta: ¿Qué usos puede tener un inmueble?

Respuesta: Los inmuebles pueden tener diferentes usos según el vocabulario KOS de EDINT: Residencial, Comercial, Industrial, etc.

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>

SELECT ?useLabel (COUNT(?property) AS ?count) WHERE {
    ?property a edintcat:CadastralProperty ;
              edintcat:hasUse ?use .

    ?use a skos:Concept ;
         skos:prefLabel ?useLabel .
}
GROUP BY ?useLabel
ORDER BY DESC(?count)

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>

SELECT ?property ?label ?participation WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?label ;
              edintcat:hasUse ?use ;
              edintcat:participation ?participation .

    ?use a skos:Concept ;
         skos:prefLabel "Residencial" .
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>

SELECT ?property ?label ?useLabel ?participation WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?label ;
              edintcat:hasUse ?use ;
              edintcat:participation ?participation .

    ?use a skos:Concept ;
         skos:prefLabel ?useLabel .

    FILTER(?useLabel IN ("Residencial", "Comercial", "Educativo"))
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>

SELECT ?useLabel (COUNT(?construction) AS ?count) WHERE {
    ?construction a edintcat:CadastralConstruction ;
                 edintcat:hasUse ?use .

    ?use a skos:Concept ;
         skos:prefLabel ?useLabel .
}
GROUP BY ?useLabel
ORDER BY DESC(?count)

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>

SELECT ?property ?label ?use ?useLabel ?schemeLabel WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?label ;
              edintcat:hasUse ?use .

    ?use a skos:Concept ;
         skos:prefLabel ?useLabel ;
         skos:inScheme ?scheme .

    ?scheme rdfs:label ?schemeLabel .
}

CAT07 - How to specify location/geometry of a parcel

Pregunta: ¿Cómo se especifica la ubicación/geometría de una parcela?

Respuesta: La ubicación se especifica mediante dirección (vocabulario callejero) y geometría (GeoSPARQL)

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX geo: <http://www.opengis.net/ont/geosparql#>
PREFIX qudt: <http://qudt.org/schema/qudt/>
PREFIX qudtunit: <http://qudt.org/vocab/unit/>
PREFIX callejero: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/callejero#>

SELECT ?parcel ?parcelLabel ?streetName ?number ?postalCode
       ?municipality ?province WHERE {
    ?parcel a edintcat:CadastralParcel ;
            rdfs:label ?parcelLabel ;
            edintcat:hasAddress ?address .

    ?address a callejero:Portal ;
             callejero:via ?street ;
             callejero:numero ?number ;
             callejero:codigoPostal ?postalCode ;
             callejero:municipio ?municipality ;
             callejero:provincia ?province .

    ?street a callejero:Via ;
             callejero:nombreVia ?streetName .
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX geo: <http://www.opengis.net/ont/geosparql#>
PREFIX qudt: <http://qudt.org/schema/qudt/>
PREFIX qudtunit: <http://qudt.org/vocab/unit/>
PREFIX callejero: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/callejero#>

SELECT ?parcel ?parcelLabel ?wkt WHERE {
    ?parcel a edintcat:CadastralParcel ;
            rdfs:label ?parcelLabel ;
            geo:hasGeometry ?geometry .

    ?geometry geo:asWKT ?wkt .
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX geo: <http://www.opengis.net/ont/geosparql#>
PREFIX qudt: <http://qudt.org/schema/qudt/>
PREFIX qudtunit: <http://qudt.org/vocab/unit/>
PREFIX callejero: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/callejero#>

SELECT ?parcel ?parcelLabel ?area WHERE {
    ?parcel a edintcat:CadastralParcel ;
            rdfs:label ?parcelLabel ;
            geo:hasGeometry ?geometry .

    ?geometry geo:hasArea ?measure .

    ?measure geo:measureAmount ?area ;
             qudt:unit qudtunit:M2 .
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX geo: <http://www.opengis.net/ont/geosparql#>
PREFIX qudt: <http://qudt.org/schema/qudt/>
PREFIX qudtunit: <http://qudt.org/vocab/unit/>
PREFIX callejero: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/callejero#>

SELECT ?property ?propertyLabel ?streetName ?number ?postalCode
       ?municipality ?province WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?propertyLabel ;
              edintcat:hasAddress ?address .

    ?address a callejero:Portal ;
             callejero:via ?street ;
             callejero:numero ?number ;
             callejero:codigoPostal ?postalCode ;
             callejero:municipio ?municipality ;
             callejero:provincia ?province .

    ?street a callejero:Via ;
             callejero:nombreVia ?streetName .
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX geo: <http://www.opengis.net/ont/geosparql#>
PREFIX qudt: <http://qudt.org/schema/qudt/>
PREFIX qudtunit: <http://qudt.org/vocab/unit/>
PREFIX callejero: <http://vocab.linkeddata.es/datosabiertos/def/urbanismo-infraestructuras/callejero#>

SELECT ?parcel ?parcelLabel ?streetName ?number ?postalCode ?municipality
       ?wkt ?area WHERE {
    ?parcel a edintcat:CadastralParcel ;

            OPTIONAL { ?parcel edintcat:hasAddress ?address } .

    ?measure geo:measureAmount ?area ;

        ?address a callejero:Portal ;

                 callejero:municipio ?municipality .
        ?street a callejero:Via ;
                 callejero:nombreVia ?streetName .
    }
}
CAT08 - What cadastral reference does a property have

Pregunta: ¿Qué referencia catastral tiene un inmueble?

Respuesta: Cada inmueble tiene una referencia catastral única compuesta por referencia de parcela (7 dígitos) y referencia de inmueble (alfanumérico con dígito de control)

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?property ?label ?parcelRef ?propertyRef ?completeRef WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?label ;
              edintcat:hasCadastralReference ?ref .

    ?ref a edintcat:CadastralReference ;
         edintcat:parcelReference ?parcelRef ;
         edintcat:propertyReference ?propertyRef .

    BIND(CONCAT(?parcelRef, ?propertyRef) AS ?completeRef)
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?property ?label ?parcelRef ?propertyRef WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?label ;
              edintcat:hasCadastralReference ?ref .

    ?ref a edintcat:CadastralReference ;
         edintcat:parcelReference ?parcelRef ;
         edintcat:propertyReference ?propertyRef .

    BIND(CONCAT(?parcelRef, ?propertyRef) AS ?completeRef)

    FILTER(?completeRef = "8942103VK2793S0001UO")
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?property ?label ?parcelRef ?propertyRef ?completeRef WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?label ;
              edintcat:hasCadastralReference ?ref .

    ?ref a edintcat:CadastralReference ;
         edintcat:parcelReference ?parcelRef ;
         edintcat:propertyReference ?propertyRef .

    BIND(CONCAT(?parcelRef, ?propertyRef) AS ?completeRef)

    FILTER(STRSTARTS(?completeRef, "8942103VK2793"))
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?parcelRef (COUNT(?property) AS ?propertyCount) WHERE {
    ?property a edintcat:CadastralProperty ;
              edintcat:hasCadastralReference ?ref .

    ?ref a edintcat:CadastralReference ;
         edintcat:parcelReference ?parcelRef ;
         edintcat:propertyReference ?propertyRef .
}
GROUP BY ?parcelRef
ORDER BY DESC(?propertyCount)

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>

SELECT ?property ?label ?parcelRef ?propertyRef ?completeRef ?participation WHERE {
    ?property a edintcat:CadastralProperty ;
              rdfs:label ?label ;
              edintcat:participation ?participation ;
              edintcat:hasCadastralReference ?ref .

    ?ref a edintcat:CadastralReference ;
         edintcat:parcelReference ?parcelRef ;
         edintcat:propertyReference ?propertyRef .

    BIND(CONCAT(?parcelRef, ?propertyRef) AS ?completeRef)
}
CAT09 - When was a construction built

Pregunta: ¿Cuándo se construyó una construcción?

Respuesta: La fecha de construcción de una construcción se almacena como año gregoriano (ej: 2012) utilizando la propiedad constructionYear

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

SELECT ?construction ?label ?constructionYear WHERE {
    ?construction a edintcat:CadastralConstruction ;
                 rdfs:label ?label ;
                 edintcat:constructionYear ?constructionYear .
}
ORDER BY ?constructionYear

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

SELECT ?construction ?label WHERE {
    ?construction a edintcat:CadastralConstruction ;
                 rdfs:label ?label ;
                 edintcat:constructionYear "2012"^^xsd:gYear .
}

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

SELECT ?construction ?label ?constructionYear WHERE {
    ?construction a edintcat:CadastralConstruction ;
                 rdfs:label ?label ;
                 edintcat:constructionYear ?constructionYear .

    FILTER(?constructionYear >= "2010"^^xsd:gYear && ?constructionYear <= "2020"^^xsd:gYear)
}
ORDER BY DESC(?constructionYear)

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

SELECT ?constructionYear (COUNT(?construction) AS ?count) WHERE {
    ?construction a edintcat:CadastralConstruction ;
                 edintcat:constructionYear ?constructionYear .
}
GROUP BY ?constructionYear
ORDER BY DESC(?constructionYear)

PREFIX edintcat: <http://vocab.linkeddata.es/datosabiertos/def/catastro/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
